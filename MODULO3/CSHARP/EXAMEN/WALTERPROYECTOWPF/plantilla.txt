Estructura del Proyecto donde cada ejemplo es un folder
--------------------------------------------------------------------------------
ProjectCSharpWPF_001/

		 MainWindow.xaml.cs
--------------------------------------------------------------------------------
MainWindow.xaml
--------------------------------------------------------------------------------
x:Class="MiAppWPF.CalculadoraSimpleButton.MainWindow" <--- Clase en el paquete (*)
xmlns:local="clr-namespace:MiAppWPF"
--------------------------------------------------------------------------------
MainWindow.xaml.cs
--------------------------------------------------------------------------------
namespace MiAppWPF.CalculadoraSimpleButton; <--- Paquete donde esta la clase (*)
public partial class MainWindow : Window    <--- Nombre de la clase (*)
public MainWindow()                         <--- Constructor de la clase (')

--------------------------------------------------------------------------------

xmlns:local="clr-namespace:MiAppWPF"  <--- Paquete principal del proyecto
StartupUri="CalculadoraSimpleButton/MainWindow.xaml"> <--- Paquete del programa (*)
